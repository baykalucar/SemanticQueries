
SELECT p.model_year, c.category_name, SUM(oi.quantity * (oi.list_price - oi.list_price * oi.discount)) AS total_sales
FROM order_items oi
JOIN products p ON oi.product_id = p.product_id
JOIN categories c ON p.category_id = c.category_id
GROUP BY p.model_year, c.category_name
ORDER BY c.category_name, p.model_year;
