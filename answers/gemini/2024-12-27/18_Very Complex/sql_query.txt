
SELECT c.customer_id, c.first_name, c.last_name, COUNT(o.order_id) AS purchase_frequency, AVG(oi.quantity * oi.list_price * (1 - oi.discount)) AS avg_order_value, GROUP_CONCAT(p.product_name) AS product_preferences FROM customers c JOIN orders o ON c.customer_id = o.customer_id JOIN order_items oi ON o.order_id = oi.order_id JOIN products p ON oi.product_id = p.product_id GROUP BY c.customer_id, c.first_name, c.last_name
